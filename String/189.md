# 189. Rotate Array

```java
// 这个写法会ot
public void rotate(int[] nums, int k) {
    while (k > 0) {
        rotatelast(nums);
        k--;
    }
}

private void rotatelast(int[] nums) {
    int temp = nums[nums.length - 1];
    for (int i = nums.length - 1; i > 0; i--) {
        nums[i] = nums[i - 1];
    }
    nums[0] = temp;
}

// 正确写法
public void rotate(int[] nums, int k) {
    int n = nums.length;
    k %= n;
    // 传下标应该是 k-1
    rotatelast(nums, 0, nums.length - 1);
    rotatelast(nums, 0, k - 1);
    rotatelast(nums, k, nums.length - 1);
}

private void rotatelast(int[] nums, int left, int right) {
    while (left < right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
        left++;
        right--;
    }
}
```
