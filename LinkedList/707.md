# 707. Design Linked List

如果你要访问节点本身（比如 get(index)） → 从 `dummy.next` 开始，或者 `i <= index`;

但如果你要修改链表（插入、删除） → 必须从 `dummy` 开始 且 `i < index`;

```java
class MyLinkedList {
    private static class Node {
        int val;
        Node next;
        Node(int val) {
            this.val = val;
        };
    }

    Node dummy;
    int size;

    public MyLinkedList() {
        dummy = new Node(0);
        size = 0;
    }

    public int get(int index) {
        if (index < 0 || index >= size)
            return -1;

        Node cur = dummy;
        for (int i = 0; i <= index; i++) {
            cur = cur.next;
        }
        return cur.val;
    }

    public void addAtHead(int val) {
        Node newNode = new Node(val);
        Node next = dummy.next;
        dummy.next = newNode;
        dummy.next.next = next;
        size++;
    }

    public void addAtTail(int val) {
        Node cur = dummy;
        while (cur.next != null) {
            cur = cur.next;
        }
        cur.next = new Node(val);
        size++;
    }

    public void addAtIndex(int index, int val) {
        // 操作的时候要在 index 的前一个进行操作。 cur.next = new;
        if (index == 0) {
            addAtHead(val);

        } else if (index == size) {
            addAtTail(val);

        } else if (index > size) {
            return;
        } else {
            Node cur = dummy;
            for (int i = 0; i < index; i++) {
                cur = cur.next;
            }
            Node temp = cur.next;
            Node newNode = new Node(val);
            cur.next = newNode;
            cur.next.next = temp;
            size++;
        }

    }

    public void deleteAtIndex(int index) {
        if (index < 0 || index >= size)
            return;
        Node cur = dummy;
        for (int i = 0; i < index; i++) {
            cur = cur.next;
        }
        cur.next = cur.next.next;
        size--;
    }
}

/**
 * Your MyLinkedList object will be instantiated and called as such:
 * MyLinkedList obj = new MyLinkedList();
 * int param_1 = obj.get(index);
 * obj.addAtHead(val);
 * obj.addAtTail(val);
 * obj.addAtIndex(index,val);
 * obj.deleteAtIndex(index);
 */
```
