```java
public ListNode detectCycle(ListNode head) {
    if (head == null || head.next == null) return null;
    ListNode fast = head;
    ListNode slow = head;

    int pos = -1;

    while (fast.next != null && fast.next.next != null) {
        fast = fast.next.next;
        slow = slow.next;
        if (fast == slow) {
            // 找环入口
            ListNode index1 = fast;
            ListNode index2 = head;
            while (index1 != index2) {
                index1 = index1.next;
                index2 = index2.next;
            }
            return index1;
        }
    }

    return null;
}
```
