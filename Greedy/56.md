# 56. Merge Intervals

```java
    public int[][] merge(int[][] intervals) {
         if (intervals == null || intervals.length == 0) return new int[0][2];
        int[][] res = new int[intervals.length][2];
        int cur = 0;
              Arrays.sort(intervals, (a, b) -> Integer.compare(a[0], b[0]));
        res[cur][0] = intervals[0][0];
        res[cur][1] = intervals[0][1];

        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] > res[cur][1]) {
                cur++;
                res[cur][0] = intervals[i][0];
                res[cur][1] = intervals[i][1];

            } else {
                //排序后 intervals[i][0] 一定 >= 当前 start ，所以这里不需要
                // res[cur][0] = Math.min(res[cur][0], intervals[i][0]);
                res[cur][1] = Math.max(intervals[i][1], res[cur][1]);
            }
        }
        return Arrays.copyOf(res, cur + 1);
    }
```
