# 1971. Find if Path Exists in Graph

```java
class Solution {
    public boolean validPath(int n, int[][] edges, int source, int destination) {
        UnionFind uf = new UnionFind(n);
        // 把所有边连起来
        for (int[] e : edges) {
            uf.union(e[0], e[1]);
        }

        // 判断是否在同一个集合
        return uf.find(source) == uf.find(destination);
    }
}

class UnionFind {
    int[] parent;

    public UnionFind(int n) {
        parent = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
        }
    }

    // // 原始 find：一路往上找根
    // public int find(int x) {
    //     while (x != parent[x]) {
    //         x = parent[x];
    //     }
    //     return x;
    // }

    // 优化后的 find：路径压缩
    public int find(int x) {
        if (x != parent[x]) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }


    // 合并两个集合
    public void union(int a, int b) {
        int pa = find(a);
        int pb = find(b);
        if (pa != pb) {
            parent[pa] = pb;
        }
    }
}
```
