# 139. Word Break

```java
    Boolean[] memo;

    public boolean wordBreak(String s, List<String> wordDict) {
        Set<String> dict = new HashSet<>(wordDict);
        memo = new Boolean[s.length()]; // 记录从每个 startIndex 开始是否可行
        return backTrack(s, dict, 0);
    }

    private boolean backTrack(String s, Set<String> dict, int startIndex) {
        if (startIndex == s.length()) return true;

        // ✅ 这个 index 算过了，直接返回，避免指数级重复
        if (memo[startIndex] != null) return memo[startIndex];

        for (int i = startIndex + 1; i <= s.length(); i++) {
            if (dict.contains(s.substring(startIndex, i))) {
                if (backTrack(s, dict, i)) {
                    memo[startIndex] = true;
                    return true;
                }
            }
        }

        return memo[startIndex] = false;
    }
```
