### LRU Cache — C# 版（你已经会 Java 版了）

```csharp
using System.Collections.Generic;

public class LRUCache {

    class Node {
        public int Key, Value;
        public Node Prev, Next;
        public Node(int k, int v) { Key = k; Value = v; }
    }

    Dictionary<int, Node> map = new Dictionary<int, Node>();
    Node head = new Node(0, 0);
    Node tail = new Node(0, 0);
    int capacity;

    public LRUCache(int capacity) {
        this.capacity = capacity;
        head.Next = tail;
        tail.Prev = head;
    }

    public int Get(int key) {
        if (!map.ContainsKey(key)) return -1;
        Node node = map[key];
        Remove(node);
        InsertHead(node);
        return node.Value;
    }

    public void Put(int key, int value) {
        if (map.ContainsKey(key)) {
            Node node = map[key];
            node.Value = value;
            Remove(node);
            InsertHead(node);
        } else {
            Node node = new Node(key, value);
            InsertHead(node);
            map[key] = node;
            if (map.Count > capacity) {
                map.Remove(tail.Prev.Key);
                Remove(tail.Prev);
            }
        }
    }

    void Remove(Node node) {
        node.Prev.Next = node.Next;
        node.Next.Prev = node.Prev;
    }

    void InsertHead(Node node) {
        node.Next = head.Next;
        node.Prev = head;
        head.Next.Prev = node;
        head.Next = node;
    }
}
```

```csharp
public bool HasCycle(ListNode head) {
    ListNode slow = head, fast = head;
    while (fast != null && fast.next != null) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow == fast) return true;

public ListNode DetectCycle(ListNode head) {
    ListNode slow = head, fast = head;
    while (fast != null && fast.next != null) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow == fast) {
            slow = head;
            while (slow != fast) {
                slow = slow.next;
                fast = fast.next;
            }
            return slow;
        }
    }
    return null;
}
```

```csharp
using System;
using System.Collections.Generic;

public int[][] Merge(int[][] intervals) {
    Array.Sort(intervals, (a, b) => a[0] - b[0]);
    List<int[]> result = new List<int[]>();
    foreach (var interval in intervals) {
        if (result.Count > 0 && result[result.Count - 1][1] >= interval[0]) {
            result[result.Count - 1][1] = Math.Max(result[result.Count - 1][1], interval[1]);
        } else {
            result.Add(new int[] { interval[0], interval[1] });
        }
    }
    return result.ToArray();
}
```

### Trapping Rain Water — C# 版（L60 2025.2 真题）

```csharp
using System;

public int Trap(int[] height) {
    int left = 0, right = height.Length - 1;
    int leftMax = 0, rightMax = 0, result = 0;
    while (left < right) {
        if (height[left] < height[right]) {
            leftMax = Math.Max(leftMax, height[left]);
            result += leftMax - height[left];
            left++;
        } else {
            rightMax = Math.Max(rightMax, height[right]);
            result += rightMax - height[right];
            right--;
        }
    }
    return result;
}
```

```csharp
public int[] TwoSum(int[] nums, int target) {
    Dictionary<int, int> map = new Dictionary<int, int>();
    for (int i = 0; i < nums.Length; i++) {
        int complement = target - nums[i];
```

|---|---|
| **Mutex vs Semaphore** | Mutex 同时只允许 1 个线程进入（厕所 1 个坑位）。Semaphore 允许 N 个（停车场 50 个车位）。 |
| **synchronized / lock** | 保证同一时间只有一个线程执行代码块。C# 用 `lock(obj){}`，Java 用 `synchronized`。 |
| **volatile** | 保证变量可见性，一个线程改了其他线程立刻看到。不保证原子性，`count++` 还是需要锁。 |
| **Producer-Consumer** | Producer 往 queue 放，Consumer 从 queue 取。用 BlockingQueue 实现。你的 JB Hi-Fi notification 系统就是这个模式。 |
| **Deadlock** | 两个线程互相等对方放锁。预防：所有线程按相同顺序获取锁。 |

### OOP（LeetCode L61 2025 + Glassdoor 确认考过）

| 问题 | 回答 |
|---|---|
| **四大特性** | 封装（隐藏实现）、继承（复用代码）、多态（同接口不同实现）、抽象（只暴露必要接口）。 |

### 场景 1：LRU Cache（线程安全版）

> LeetCode 146 你已经会写代码了。如果面试官追问并发，口述以下要点即可。

**口述要点：**
- 数据结构：`Dictionary` 做 O(1) 查找 + 双向链表维护访问顺序
- Get/Put 都要移动链表节点，所以整个操作要用 `lock` 包起来
- 为什么不用 `ReaderWriterLock`？→ 因为 Get 也修改链表，读写都需要互斥
- 加 TTL？→ Node 加 `ExpireAt` 字段，Get 时检查过期就删掉返回 -1
- 优化？→ 分段锁（按 key 的 hash 分到不同锁），减少锁竞争

**C# 代码（线程安全 LRU Cache）：**

```csharp
using System;
using System.Collections.Generic;
using System.Threading;


// 定义 eviction 接口
public interface IEvictionPolicy {
    void OnAccess(int key);
    void OnInsert(int key);
    int Evict();
}

// LRU 实现
public class LRUPolicy : IEvictionPolicy {
    private LinkedList<int> _list = new();
    private Dictionary<int, LinkedListNode<int>> _map = new();

    public void OnAccess(int key) {
        if (_map.ContainsKey(key)) {
            _list.Remove(_map[key]);
            _map[key] = _list.AddFirst(key);
        }
    }

    public void OnInsert(int key) {
        var node = _list.AddFirst(key);
        _map[key] = node;
    }

    public int Evict() {
        var last = _list.Last.Value;
        _list.RemoveLast();
        _map.Remove(last);
        return last;
    }
}

public class LRUCache {
    private int _capacity;
    private Dictionary<int, int> _store = new();
    private IEvictionPolicy _policy;

    public LRUCache(int capacity,IEvictionPolicy policy) {
        _capacity = capacity;
        _policy = policy;
    }

    public int Get(int key) {
        if (!_store.ContainsKey(key)) return -1;
        _policy.OnAccess(key);
        return _store[key];
    }

    public void Put(int key, int value) {
        if (_store.ContainsKey(key)) {
            _store[key] = value;
            _policy.OnAccess(key);
        } else {
            if (_store.Count == _capacity) {
                int evicted = _policy.Evict();
                _store.Remove(evicted);
            }
            _store[key] = value;
            _policy.OnInsert(key);
        }
    }
}
```

### 场景 2：Thread Pool（线程池）

> GFG L60 2025.12 真题。不会让你现场写完整代码，口述思路即可。

**口述要点：**
- 三个核心组件：**任务队列（BlockingQueue）+ 一组 Worker 线程 + Submit 方法**
- 工作原理：调用方 Submit 任务到队列，Worker 线程不断从队列取任务执行
- 为什么用线程池不用每次 new Thread？→ 创建/销毁线程开销大，线程池复用线程
- 队列满了怎么办？→ 设 `boundedCapacity`，满了 Submit 会阻塞等待
- 关闭线程池？→ 标记不再接受新任务（`CompleteAdding`），等所有 Worker 完成当前任务后退出
- C# 现成的？→ `ThreadPool` 类 和 `Task.Run()` 就是封装好的线程池

**C# 代码（简易线程池）：**

```csharp
using System;
using System.Collections.Concurrent;
using System.Threading;

public class SimpleThreadPool : IDisposable {
    private readonly BlockingCollection<Action> _taskQueue;
    private readonly Thread[] _workers;
    private bool _disposed;

    public SimpleThreadPool(int workerCount, int maxQueueSize = 100) {
        _taskQueue = new BlockingCollection<Action>(maxQueueSize);
        _workers = new Thread[workerCount];

        for (int i = 0; i < workerCount; i++) {
            _workers[i] = new Thread(WorkerLoop) { IsBackground = true };
            _workers[i].Start();
        }
    }

    private void WorkerLoop() {
        foreach (var task in _taskQueue.GetConsumingEnumerable()) {
            try { task(); }
            catch (Exception ex) {
                Console.WriteLine($"Task error: {ex.Message}");
            }
        }
    }

    public void Submit(Action task) {
        if (_disposed) throw new InvalidOperationException("Pool is shut down");
        _taskQueue.Add(task); // 队列满时自动阻塞
    }

    public void Shutdown() {
        _taskQueue.CompleteAdding(); // 不再接受新任务
        foreach (var w in _workers) w.Join(); // 等所有 worker 完成
    }

    public void Dispose() {
        if (!_disposed) {
            Shutdown();
            _taskQueue.Dispose();
            _disposed = true;
        }
    }
}
```

### 场景 3：Rate Limiter（限流器）

> 微软高频 LLD 题。口述 Token Bucket 算法即可。

**口述要点：**
- 算法：**Token Bucket（令牌桶）**
- 桶里最多 N 个令牌，每秒按 refillRate 自动补充
- 每个请求消耗 1 个令牌，没令牌就拒绝（return false）
- 核心操作：先算经过了多少时间 → 补充令牌（不超过上限）→ 扣减 1 个
- 并发：整个 Refill + 扣减 要用 `lock` 包起来，因为是两步操作
- 多用户？→ 用 `ConcurrentDictionary<userId, RateLimiter>`，每个用户一个桶
- Token Bucket vs Sliding Window？→ Token Bucket 允许短时间突发，Sliding Window 更严格均匀
- 分布式？→ 令牌数存 Redis，用 Lua 脚本保证原子性

**C# 代码（Token Bucket 限流器）：**

```csharp
using System;
using System.Collections.Concurrent;

public class TokenBucketRateLimiter {
    private readonly int _maxTokens;
    private readonly double _refillPerSecond;
    private double _tokens;
    private DateTime _lastRefill;
    private readonly object _lock = new();

    public TokenBucketRateLimiter(int maxTokens, double refillPerSecond) {
        _maxTokens = maxTokens;
        _refillPerSecond = refillPerSecond;
        _tokens = maxTokens;
        _lastRefill = DateTime.UtcNow;
    }

    public bool TryConsume() {
        lock (_lock) {
            Refill();
            if (_tokens >= 1) {
                _tokens -= 1;
                return true;
            }
            return false;
        }
    }

    private void Refill() {
        var now = DateTime.UtcNow;
        var elapsed = (now - _lastRefill).TotalSeconds;
        _tokens = Math.Min(_maxTokens, _tokens + elapsed * _refillPerSecond);
        _lastRefill = now;
    }
}

// 多用户版本
public class MultiUserRateLimiter {
    private readonly ConcurrentDictionary<string, TokenBucketRateLimiter> _limiters = new();
    private readonly int _maxTokens;
    private readonly double _refillPerSecond;

    public MultiUserRateLimiter(int maxTokens, double refillPerSecond) {
        _maxTokens = maxTokens;
        _refillPerSecond = refillPerSecond;
    }

    public bool TryConsume(string userId) {
        var limiter = _limiters.GetOrAdd(userId,
            _ => new TokenBucketRateLimiter(_maxTokens, _refillPerSecond));
        return limiter.TryConsume();
    }
}
```

### 场景 4：Booking System（预订系统 / 防 Overbooking）

> 你第一轮已经考过了。口述两种方案对比。

**口述要点：**

**方案 A — 悲观锁：**
- 第一个人来 → `lock` 锁住 → 检查有没有空位 → 有就扣减 → 释放锁
- 优点：简单可靠，不会超卖
- 缺点：并发性能差，所有人排队
- 适合：冲突频繁（抢票、秒杀）
- 数据库层面：`SELECT ... FOR UPDATE`

**方案 B — 乐观锁（CAS）：**
- 读当前值 → 计算新值 → 提交时检查值有没有被别人改过 → 改过就重试
- 优点：不阻塞，并发性能好
- 缺点：冲突多时反复重试浪费资源
- 适合：冲突少（普通预订）
- 数据库层面：`WHERE version = @oldVersion`

**追问：**
- 超时？→ 预订 15 分钟不付款自动释放
- 部分失败？→ 事务（Transaction），要么全成功要么全回滚

**C# 代码（悲观锁 + 乐观锁 两种方案）：**

```csharp
using System;
using System.Threading;

// 方案 A：悲观锁
public class PessimisticBookingSystem {
    private int _availableSeats;
    private readonly object _lock = new();

    public PessimisticBookingSystem(int totalSeats) {
        _availableSeats = totalSeats;
    }

    public bool TryBook(int count) {
        lock (_lock) {
            if (_availableSeats >= count) {
                _availableSeats -= count;
                return true;
            }
            return false;
        }
    }

    public void Cancel(int count) {
        lock (_lock) {
            _availableSeats += count;
        }
    }
}

// 方案 B：乐观锁（CAS）
public class OptimisticBookingSystem {
    private int _availableSeats;

    public OptimisticBookingSystem(int totalSeats) {
        _availableSeats = totalSeats;
    }

    public bool TryBook(int count) {
        while (true) {
            int current = _availableSeats;
            if (current < count) return false;

            // CAS：如果当前值没被别人改过，才更新
            int original = Interlocked.CompareExchange(
                ref _availableSeats, current - count, current);

            if (original == current) return true;
            // 被别人改了，重试
        }
    }

    public void Cancel(int count) {
        Interlocked.Add(ref _availableSeats, count);
    }
}
```

### 场景 5：Producer-Consumer Queue（生产者-消费者）

> 跟你 JB Hi-Fi 的 notification 系统直接相关，口述即可。

**口述要点：**
- Producer 往队列放消息，Consumer 从队列取消息处理
- 队列满了 → Producer 阻塞等待（`Monitor.Wait`）
- 队列空了 → Consumer 阻塞等待（`Monitor.Wait`）
- 有新消息/有空位 → `Monitor.PulseAll` 通知对方
- 为什么用 `while` 不用 `if` 检查条件？→ 防止虚假唤醒（Spurious Wakeup）
- C# 现成的？→ `BlockingCollection<T>`，设 `boundedCapacity`，`Add` 满了会阻塞，`Take` 空了会阻塞
- 跟你的项目？→ "JB Hi-Fi 的 notification 系统就是这个模式，订单服务是 Producer，notification 服务是 Consumer，中间用 message queue 解耦"

**C# 代码（Producer-Consumer + Monitor）：**

```csharp
using System.Collections.Generic;
using System.Threading;

public class BoundedQueue<T> {
    private readonly Queue<T> _queue = new();
    private readonly int _capacity;
    private readonly object _lock = new();

    public BoundedQueue(int capacity) {
        _capacity = capacity;
    }

    public void Produce(T item) {
        lock (_lock) {
            while (_queue.Count >= _capacity)
                Monitor.Wait(_lock); // 队列满，等待

            _queue.Enqueue(item);
            Monitor.PulseAll(_lock); // 通知 Consumer 有新数据
        }
    }

    public T Consume() {
        lock (_lock) {
            while (_queue.Count == 0)
                Monitor.Wait(_lock); // 队列空，等待

            T item = _queue.Dequeue();
            Monitor.PulseAll(_lock); // 通知 Producer 有空位
            return item;
        }
    }
}

// 更简洁的 C# 内置写法：
// var queue = new BlockingCollection<Task>(boundedCapacity: 100);
// Producer: queue.Add(task);        ← 满了自动阻塞
// Consumer: queue.Take();           ← 空了自动阻塞
// 关闭:    queue.CompleteAdding();  ← 不再接受新任务
```

### 场景 6：Multi-Level Parking Lot（多层停车场）

> LeetCode Discuss 2026.3 微软 SWE-II 真题。这道偏 OOP 设计，口述类和关系。

**口述要点 — 核心类（5 个）：**
- **Vehicle**：有 licensePlate 和 type（Motorcycle / Car / Bus）
- **ParkingSpot**：有 spotId、type（Motorcycle / Compact / Large）、isAvailable。有 `CanFit(vehicle)` 方法判断能不能停
- **ParkingFloor**：有 floorNumber 和一组 spots，有 `FindAvailableSpot(vehicle)` 方法
- **Ticket**：记录 vehicle、spot、entryTime、exitTime，有 `CalculateFee()` 方法
- **ParkingLot**：核心类，有多个 floor + activeTickets 字典。`Entry()` 遍历每层找空位，`Exit()` 算费释放车位

**车位匹配规则：**
- 摩托车 → 任何车位都能停
- 小车 → Compact 或 Large，不能停 Motorcycle 位
- 大车 → 只能停 Large

**并发：** Entry/Exit 用 `lock` 保证同一时间只有一辆车在操作

**追问：**
- 优化查找？→ 每层维护按类型分组的可用车位计数器，O(1) 判断有没有空位
- 不同计费策略？→ Strategy Pattern，抽出 `IFeeCalculator` 接口
- 电动车充电？→ 新增 `SpotType.Electric`，Spot 加 `HasCharger` 属性

**OOP 考点：** 封装（各类各司其职）、多态（CanFit 按车型不同逻辑）、SOLID（单一职责）、Strategy Pattern（计费）

**C# 代码（多层停车场 OOP 设计）：**

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public enum VehicleType { Motorcycle, Car, Bus }
public enum SpotType { Motorcycle, Compact, Large }

public class Vehicle {
    public string LicensePlate { get; }
    public VehicleType Type { get; }
    public Vehicle(string plate, VehicleType type) {
        LicensePlate = plate;
        Type = type;
    }
}

public class ParkingSpot {
    public int SpotId { get; }
    public SpotType Type { get; }
    public bool IsAvailable { get; private set; } = true;

    public ParkingSpot(int id, SpotType type) { SpotId = id; Type = type; }

    public bool CanFit(Vehicle v) => v.Type switch {
        VehicleType.Motorcycle => true,               // 摩托车什么位都能停
        VehicleType.Car => Type != SpotType.Motorcycle, // 小车不能停摩托车位
        VehicleType.Bus => Type == SpotType.Large,      // 大车只能停 Large
        _ => false
    };

    public void Park() => IsAvailable = false;
    public void Free() => IsAvailable = true;
}

public class ParkingFloor {
    public int FloorNumber { get; }
    private readonly List<ParkingSpot> _spots;

    public ParkingFloor(int floor, List<ParkingSpot> spots) {
        FloorNumber = floor;
        _spots = spots;
    }

    public ParkingSpot FindAvailable(Vehicle v) {
        return _spots.FirstOrDefault(s => s.IsAvailable && s.CanFit(v));
    }
}

public class Ticket {
    public Vehicle Vehicle { get; }
    public ParkingSpot Spot { get; }
    public DateTime EntryTime { get; }
    public DateTime? ExitTime { get; private set; }

    public Ticket(Vehicle v, ParkingSpot spot) {
        Vehicle = v; Spot = spot; EntryTime = DateTime.UtcNow;
    }

    public decimal CalculateFee(decimal ratePerHour) {
        ExitTime = DateTime.UtcNow;
        var hours = Math.Ceiling((ExitTime.Value - EntryTime).TotalHours);
        return (decimal)hours * ratePerHour;
    }
}

public class ParkingLot {
    private readonly List<ParkingFloor> _floors;
    private readonly Dictionary<string, Ticket> _activeTickets = new();
    private readonly object _lock = new();

    public ParkingLot(List<ParkingFloor> floors) { _floors = floors; }

    public Ticket Entry(Vehicle vehicle) {
        lock (_lock) {
            foreach (var floor in _floors) {
                var spot = floor.FindAvailable(vehicle);
                if (spot != null) {
                    spot.Park();
                    var ticket = new Ticket(vehicle, spot);
                    _activeTickets[vehicle.LicensePlate] = ticket;
                    return ticket;
                }
            }
            return null; // 没有空位
        }
    }

    public decimal Exit(string licensePlate, decimal ratePerHour) {
        lock (_lock) {
            if (!_activeTickets.TryGetValue(licensePlate, out var ticket))
                throw new Exception("Vehicle not found");

            var fee = ticket.CalculateFee(ratePerHour);
            ticket.Spot.Free();
            _activeTickets.Remove(licensePlate);
            return fee;
        }
    }
}
```
