# 98. Validate Binary Search Tree

```java
// The better way is to use private long prev = Long.MIN_VALUE;
// Perform in-order traversal while comparing values
private List<Integer> list;

public boolean isValidBST(TreeNode root) {
    if (root == null) return true;
    list = new ArrayList<>();
    dfs(root);
    boolean flag = true;
    for (int i = 0; i < list.size() - 1; i++) {
        if (list.get(i) >= list.get(i + 1)) {
            flag = false;
            break;
        }
    }
    return flag;
}

private void dfs(TreeNode root) {
    if (root == null) return;

    dfs(root.left);
    list.add(root.val);
    dfs(root.right);
}
```
