# 112. Path Sum

```java
    private List<List<Integer>> list;
    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        list = new ArrayList<>();

        if (root == null)   return list;

        dfs(root, targetSum, 0, new ArrayList<>() );
        return list;
    }

    private void dfs(TreeNode root, int targetSum, int cur,List<Integer> path) {
        if (root == null) return;
        int curLength = path.size();

        path.add(root.val);
        if (root.left == null && root.right == null && cur + root.val == targetSum){
     // ⭐ 必须做拷贝
            list.add(new ArrayList<>(path));
        }
        dfs(root.left, targetSum, cur + root.val,path);
        dfs(root.right, targetSum, cur + root.val,path);
        path.remove(curLength);
    }
```
