# 112. Path Sum

```java
public boolean hasPathSum(TreeNode root, int targetSum) {
    if (root == null) return false;
    return dfs(root, targetSum, 0);
}

private boolean dfs(TreeNode root, int targetSum, int cur) {
    if (root == null) {
        return false;
    }
    if (root.left == null && root.right == null && cur + root.val == targetSum) {
        return true;
    }
    boolean left = dfs(root.left, targetSum, cur + root.val);
    boolean right = dfs(root.right, targetSum, cur + root.val);

    return left || right;
}
```
