# 130. Surrounded Regions

```java
class Solution {
    boolean flag = false;
    int[][] dirs = {{0,1},{1,0},{-1,0},{0,-1}};

    public void solve(char[][] board) {
        int n = board.length;
        int m = board[0].length;
        if (n == 0 || m == 0) return;

        boolean[][] visited = new boolean[n][m];

        // 1) 先把边界连通的 'O' 标记 visited（不翻）
        for (int i = 0; i < n; i++) {
            if (board[i][0] == 'O' && !visited[i][0]) dfs(board, visited, i, 0);
            if (board[i][m - 1] == 'O' && !visited[i][m - 1]) dfs(board, visited, i, m - 1);
        }
        for (int j = 0; j < m; j++) {
            if (board[0][j] == 'O' && !visited[0][j]) dfs(board, visited, 0, j);
            if (board[n - 1][j] == 'O' && !visited[n - 1][j]) dfs(board, visited, n - 1, j);
        }

        // 2) 再把内部未 visited 的 'O' 全部翻成 'X'
        flag = true;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (board[i][j] == 'O' && !visited[i][j]) {
                    dfs(board, visited, i, j);
                }
            }
        }
    }

    private void dfs(char[][] board, boolean[][] visited, int x, int y) {
        visited[x][y] = true;
        if (flag) board[x][y] = 'X';

        for (int i = 0; i < 4; i++) {
            int nx = x + dirs[i][0];
            int ny = y + dirs[i][1];

            if (nx < 0 || nx >= board.length || ny < 0 || ny >= board[0].length) continue;

            if (board[nx][ny] == 'O' && !visited[nx][ny]) {
                dfs(board, visited, nx, ny);
            }
        }
    }
}
```
