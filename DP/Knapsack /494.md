# 494. Target Sum

```java

int way = 0; 
int cur = 0; 

public int findTargetSumWays(int[] nums, int target) {
    backTrack(0, nums, target);
    return way;
}

private void backTrack(int index, int[] nums, int target) {

    if (index == nums.length) {

        if (cur == target) {
            way++;
        }
        return; 
    }

    // 选择加当前数字
    cur = cur + nums[index]; // 加上当前数字
    backTrack(index + 1, nums, target); // 递归到下一个数字
    cur = cur - nums[index]; // 回溯，撤销加法操作

    // 选择减当前数字
    cur = cur - nums[index]; // 减去当前数字
    backTrack(index + 1, nums, target); // 递归到下一个数字
    cur = cur + nums[index]; // 回溯，撤销减法操作
}
```
