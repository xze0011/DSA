# 213. House Robber II

```java
class Solution {
    public int rob(int[] nums) {
        int n = nums.length;
        if (n == 0) return 0;
        if (n == 1) return nums[0];

        //左闭右开
        return Math.max(robraw(nums, 0, n - 1), robraw(nums, 1, n));
    }

    private int robraw(int[] nums, int start, int end) {
        if (end - start == 1) return nums[start]; // ✅ 补丁：区间只有一间

        int n = nums.length;
        int[] dp = new int[n];

        dp[start] = nums[start];
        dp[start + 1] = Math.max(nums[start], nums[start + 1]);

        for (int i = start + 2; i < end; i++) {
            dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
        }

        return dp[end - 1];
    }
}
```
