    public int maxProfit(int[] prices) {
        int n = prices.length;
        int[][] dp = new int[n][4];
        dp[0][0] = -prices[0];
        dp[0][1] = 0;
        dp[0][2] =  -prices[0];
        dp[0][3] = 0;

        // dp[i][0] 表示：在第 i 天结束时，
        //                第一次手里【持有一支股票】的情况下，
        //                能获得的【最大利润】
        //
        // dp[i][1] 表示：在第 i 天结束时，
        //                第一次手里【不持有股票】的情况下，
        //                能获得的【最大利润】

        // dp[i][2] 表示：在第 i 天结束时，
        //                第二次手里【持有一支股票】的情况下，
        //                能获得的【最大利润】

        // dp[i][3] 表示：在第 i 天结束时，
        //                第二次手里【不持有一支股票】的情况下，
        //                能获得的【最大利润】

        for (int i = 1; i < n; i++) {
            //  因为可以多次买卖，前一天不持有，今天第一天持有，即  dp[i - 1][1] - prices[i];
            dp[i][0] = Math.max(dp[i - 1][0], -prices[i]);
            dp[i][1] = Math.max(dp[i - 1][1], dp[i - 1][0] + prices[i]);
            dp[i][2] = Math.max(dp[i - 1][2], dp[i - 1][1] - prices[i]);
            dp[i][3] = Math.max(dp[i - 1][3], dp[i - 1][2] + prices[i]);
        }
        //dp[n-1][3] ≥ dp[n-1][1]
        return dp[n-1][3];
    }
}
