# 337. House Robber III

```java
    public int rob(TreeNode root) {
        if (root == null)
            return 0;
        int[] res = new int[2];
        res = dfs(root);
        return Math.max(res[0], res[1]);
    }

    private int[] dfs(TreeNode root) {
        if (root == null) {
            return new int[] { 0, 0 };
        }
        int[] leftNode = dfs(root.left);
        int[] rightNode = dfs(root.right);

        //不偷当前root
        int val2 = Math.max(leftNode[0], leftNode[1]) + Math.max(rightNode[0], rightNode[1]);

        //偷当前root
        int val1 = root.val + leftNode[0] + rightNode[0];

        // 下标0：不偷，下标1：偷
        return new int[] { val2, val1 };
    }
```
