# 1152. Analyze User Website Visit Pattern

```java
import java.util.*;

class Solution {
    public List<String> mostVisitedPattern(String[] username, int[] timestamp, String[] website) {
        ArrayList<int[]> timeStampList = new ArrayList<>();
        HashMap<String, List<String>> map = new HashMap<>();
        HashMap<String, Integer> resMap = new HashMap<>();

        for (int i = 0; i < username.length; i++) {
            timeStampList.add(new int[] { timestamp[i], i });
        }
        Collections.sort(timeStampList, (a, b) -> a[0] - b[0]);

        // ⭐ 修复：每个用户独立 list
        for (int i = 0; i < username.length; i++) {
            int index = timeStampList.get(i)[1];
            map.putIfAbsent(username[index], new ArrayList<>());
            map.get(username[index]).add(website[index]);

        }

        // ⭐ 拼接完整访问路径
        for (String name : map.keySet()) {
            List<String> list = map.get(name);
            HashSet<String> seen = new HashSet<>();
            int n = list.size();
            for (int i = 0; i < n - 2; i++) {
                for (int j = i + 1; j < n - 1; j++) {
                    for (int k = j + 1; k < n; k++) {
                        String key = list.get(i) + '-' + list.get(j) + '-' + list.get(k);
                        if (!seen.contains(key)) {
                            resMap.put(key, resMap.getOrDefault(key, 0) + 1);
                        }
                        seen.add(key);
                    }
                }
            }
        }

        // ⭐ 找最高频路径
        String best = "";
        int max = 0;

        for (String path : resMap.keySet()) {
            int freq = resMap.get(path);

            if (freq > max || (freq == max && path.compareTo(best) < 0)) {
                max = freq;
                best = path;
            }
        }

        return Arrays.asList(best.split("-"));
    }
}
```
